15   SERVICE_TYPES = [
16       "Garbage Collection",
17       "Water Supply",
18       "Electricity Repair",
19       "Road Maintenance",
20       "Public Safety",
21       "Appoint Public Meetings"

24   # Class to represent a single service request
25   class ServiceRequest:
26       def __init__(self, request_id, name, address, service_type, description=None):
27           self.request_id = request_id
28           self.name = name
29           self.address = address
30           self.service_type = service_type
31           self.description = description
32           self.status = ServiceStatus.PENDING  # Default status is Pending
33   
34       def __str__(self):
35           return f"""
36           Request ID: {self.request_id}
37           Name: {self.name}
38           Address: {self.address}
39           Service Type: {self.service_type}
40           Description: {self.description or 'No description provided'}
41           Status: {self.status.value}
42           """

44   # Class to manage all client service requests
45   class CityService:
46       def __init__(self):
47           self.requests = []
48   
49       def generate_request_id(self):
50           """ Generate a unique request ID """
51           return str(uuid.uuid4())[:8]  # Generates an 8-character unique ID
52   
53       def add_request(self, name, address, service_type, description=None):
54           """ Add a new service request """
55           request_id = self.generate_request_id()
56           new_request = ServiceRequest(request_id, name, address, service_type, description)
57           self.requests.append(new_request)
58           print(f"\nRequest added. Your Request ID: {request_id}")
59   
60       def view_pending_requests(self):
61           """ Display all pending service requests """
62           pending_requests = [request for request in self.requests if request.status == ServiceStatus.PENDING]
63           if not pending_requests:
64               print("\nNo pending service requests found.")
65           else:
66               print("\nPending Service Requests:")
67               for request in pending_requests:
68                   print(request)
